template <typename T>

int find(T arrayobj[] ,int rollno,int n)
{
	int flag=0,index=0;
	for(int i=0;i<n;i++)
	{
		if(arrayobj[i].rno==rollno)
		{	flag=1;
			index=i;
			break;
		}
	}
	if(flag==0)
		return -1;
	else
		return index;
}
