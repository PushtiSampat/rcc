template <typename T>

void Bubble_sort(T arr[],int n) //Non-Type Argument
{
	int i,j;
	T temp;
	for(i=0;i<n;i++)
	{
		for(j=0;j<n-i-1;j++)
		{
			if(arr[j]>arr[j+1])
			{
				temp=arr[j];
				arr[j]=arr[j+1];
				arr[j+1]=temp;
			}
		}
	}
	cout<<"\n Sorted Completed.";
}
